CREATE TABLE roles
(
  id   int         NOT NULL
    PRIMARY KEY,
  role VARCHAR(20) NOT NULL
);

CREATE TABLE users
(
  id int PRIMARY KEY AUTO_INCREMENT,
  username VARCHAR(40) NOT NULL,
  password VARCHAR(15) NOT NULL,
  role_id  INT         NOT NULL,
  CONSTRAINT users_roles_id_fk
  FOREIGN KEY (role_id) REFERENCES roles (id)
    ON UPDATE CASCADE
    ON DELETE CASCADE
);

CREATE TABLE race
(
  id   INT AUTO_INCREMENT
    PRIMARY KEY,
  race VARCHAR(30) NOT NULL
);

CREATE TABLE rate_result
(
  id int PRIMARY KEY,
  result VARCHAR(20) NULL
);

CREATE TABLE rate
(
  id        INT AUTO_INCREMENT
    PRIMARY KEY,
  race_id   INT         NOT NULL,
  rate      VARCHAR(50) NOT NULL,
  result_id INT         NOT NULL,
  CONSTRAINT race___fk
  FOREIGN KEY (race_id) REFERENCES race (id)
    ON UPDATE CASCADE
    ON DELETE CASCADE,
  CONSTRAINT result___fk
  FOREIGN KEY (result_id) REFERENCES rate_result (id)
    ON UPDATE CASCADE
    ON DELETE CASCADE
);
 
CREATE TABLE user_rates
(
  user_id INT NOT NULL,
  rate_id INT NOT NULL,
  CONSTRAINT user_rates_users_id_fk
  FOREIGN KEY (user_id) REFERENCES users (id)
    ON UPDATE CASCADE
    ON DELETE CASCADE,
  CONSTRAINT user_rates_rate_id_fk
  FOREIGN KEY (rate_id) REFERENCES rate (id)
    ON UPDATE CASCADE
    ON DELETE CASCADE
);
